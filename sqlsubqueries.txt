SELECT e.employee_id,
       e.employee_name,
       e.department_id,
       e.salary
FROM Employees e
WHERE e.salary >
      (
          SELECT AVG(e2.salary)
          FROM Employees e2
          WHERE e2.department_id = e.department_id
      );

SELECT employee_id,
       employee_name,
       department_id,
       salary
FROM Employees
WHERE department_id =
(
    SELECT department_id
    FROM Employees
    GROUP BY department_id
    ORDER BY AVG(salary) DESC
    LIMIT 1
);
