
SELECT EMP_NAME,SALARY
FROM EMPLOYEES
WHERE SALARY>40000;

SELECT EMP_NAME, CITY
FROM EMPLOYEES
WHERE EMP_NAME LIKE 'a%';

SELECT EMP_NAME,DEPARTMENT
FROM EMPLOYEES
WHERE DEPARTMENT  IN ('HR', 'FINANCE');

SELECT EMP_NAME,DEPARTMENT
FROM EMPLOYEES
WHERE DEPARTMENT IS  NULL;

SELECT COUNT(*)
FROM EMPLOYEES
WHERE SALARY>30000;

select AVG(SALARY)
FROM EMPLOYEES
WHERE  DEPARTMENT='IT';

SELECT DEPARTMENT,SUM(SALARY)
FROM EMPLOYEES
WHERE CITY='HYD'
GROUP BY DEPARTMENT;

SELECT DEPARTMENT,SUM(SALARY) AS TOTAL_DEP_SALARY
FROM EMPLOYEES
GROUP BY DEPARTMENT;

SELECT DEPARTMENT,COUNT(DEPARTMENT)
FROM EMPLOYEES
GROUP BY DEPARTMENT;

SELECT EMP_NAME,DEPARTMENT,AVG(SALARY)
FROM EMPLOYEES
GROUP BY DEPARTMENT,EMP_NAME;

FROM EMPLOYEES
